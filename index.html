// 1. Constants (configure once at onboarding)
TENANT_ID     = "mewa-tenant"          // from Geotab IT
USER_ID       = "mewa-api-user"        // from Geotab IT
ORDER_TYPE_ID = "abc123"               // from GET /orderTypes
BRANCH_MAP    = { "g01":"MEWA Manching","g03":"MEWA Rodgau","g04":"MEWA Bottrop",... } // 23 branches — see CONFIG

// 2. Per Tour from Midsys:
order = {
  id:          tour.tourId,
  sourceType:  "API",                  // always fixed
  orderStatus: 1,                      // always fixed
  sourceSystem:"Midsys",
  orderType:   { id: ORDER_TYPE_ID },
  account:     { id: tour.customerId },
  appointments: tour.patches.map(patch => ({
    id:               patch.patchId,
    branch:           { id: BRANCH_MAP[patch.territory] },
    orderType:        { id: ORDER_TYPE_ID },
    appointmentStatus:1,
    contactDetails: {
      name:                    patch.customerName,
      primaryPhone:            patch.phone,
      preferredContactMethod:  "phone",
      doNotPhone:false, doNotEmail:false, doNotText:false
    },
    address: {
      fullAddress:  patch.fullAddress,
      street1:      patch.street1,
      city:         patch.city,
      postalCode:   patch.plz,
      countryCode:  "DE",
      geoCode:      { latitude: patch.lat, longitude: patch.lng }
    },
    arrivalDate:        patch.date,           // "YYYY-MM-DD"
    arrivalWindowBegin: patch.windowStart,         // "HH:MM"
    arrivalWindowEnd:   patch.windowEnd,
    customerId:         patch.customerId,
    appointmentNotes: patch.approachNote ? [{
      comment:          patch.approachNote,
      noteSource:       "Midsys",
      createUser:       "midsys-sync",
      createDateTimeUtc:new Date().toISOString()
    }] : [],
    billOfMaterials: patch.dnItems.map(item => ({
      id:                        item.itemId,
      name:                      item.patchNr + " | " + item.description,
      quantity:                  item.quantity,
      unitOfMeasurement:         item.unit,
      perUnitValue:              item.weight ?? 0,
      perUnitCost:               0,
      perUnitRevenue:            0,
      currency:                  "USD",        // ⚠ EUR TBC
      transferType:              "DELIVERY",
      hazmat:                    false,
      durationModifierPerQuantity:0
    }))
  }))
}

// ── Phase 7: Status-Tracking ──────────────────────────────────
// Visit-Status Enum (aus Swagger):
// 1 = Assigned    — Ressource zugewiesen, noch nicht gestartet
// 2 = Enroute     — Fahrer unterwegs (driver-app ODER GPS auto)
// 3 = InProgress  — Fahrer am Standort, Job läuft
// 4 = Completed   — Job fertig, Fahrer hat abgeschlossen
// 5 = Cancelled   — Storniert
// 6 = NotDone     — Nicht erledigt (z.B. Kunde nicht da)
// 7 = Undone      — Rückgängig gemacht
// 8 = Unassigned  — Noch keine Ressource zugewiesen

// 7a: Status polling — alle 5-15 Min während Dispatch-Tag
function pollVisitStatus(branchId, date, lastPollTs):
  schedules = GET("/dispatch/branchSchedules/" + branchId, { date: date })
  changes = GET("/dispatch/scheduleChanges", { branchId, since: lastPollTs })

  for change in changes:
    visit = change.visit
    switch visit.visitStatus:
      case 2:  // Enroute
        midsys.updateStatus(visit.appointmentId, "UNTERWEGS",
          visit.planned.enrouteDateTimeUtc)
      case 3:  // InProgress — Ankunft
        midsys.updateStatus(visit.appointmentId, "AM_KUNDEN",
          visit.actuals.onsiteArrivalDateTimeUtc)
      case 4:  // Completed — Abschluss
        midsys.updateStatus(visit.appointmentId, "ERLEDIGT",
          visit.actuals.closeDateTimeUtc)
        midsys.updateSequence(visit.appointmentId,
          visit.actuals.sequence)    // tatsächliche Reihenfolge
      case 6:  // NotDone
        midsys.updateStatus(visit.appointmentId, "NICHT_ERLEDIGT",
          now(), visit.actuals.closeCodesComment)

      if visit.actuals.formTemplateSubmission:
        createCRMServiceTicket({ appointmentId, ...visit.actuals })
    case 6:  // NotDone
      CRM.updateJob(appointmentId, "NICHT_ERLEDIGT",
        visit.actuals.closeCodesComment)
    case 8:  // Still Unassigned after engine attempt
      CRM.flagForManualDispatch(appointmentId)
      // → dispatcher sees in Dispatch view, assigns manually

function dailySync():
  ensureAuth()                              // Phase 0: Token prüfen

  for date in next8Days():
    midsysData = fetchFromMidsys(date)

    byBranch = groupBy(midsysData.touren, t => t.gesellschaftCode)

    for gesellschaftCode, touren in byBranch:
      branchId  = CONFIG.BRANCH_MAP[gesellschaftCode]
      resources = GET("/admin/resources?branchId=" + branchId)

      // Phase 2: Customer-Stammdaten — nur geänderte Kunden senden
      changedCustomers = touren
        .flatMap(t => t.patches)
        .filter(p => customerChangedSinceLastSync(p.kundennummer))
        .map(p => getCustomerFromERP(p.kundennummer))
      syncCustomers(changedCustomers)

      // Phase 3: Baseline-Plan — falls Midsys Ist-Zeiten liefert
      // Phase 4: PUT /orders — falls nur Planzeiten vorhanden
      if midsysData.hasActualTimes:
        planId = createBaselinePlan(date, branchId, touren, resources)
        // Phase 3 enthält status:"DISPATCH" → sofort im Dispatch-View
      else:
        PUT("/orders", buildOrders(touren, date))
        // Dispatcher plant manuell oder via Optimierung

      // Validierung: Anzahl Appointments im System prüfen
      result = GET("/orders?appointmentBranchId=" + branchId
                             + "&appointmentArrivalDate=" + date)
      expected = touren.flatMap(t => t.patches).length
      if result.appointments.length != expected:
        alert("Sync-Fehler: " + branchId + " " + date)

  log("Daily sync completed " + now())

API Terms
TermContextDescription
tenant-id HTTP Header — all requests Unique MEWA tenant name in the Advanced Routing system. Provisioned by Geotab. Required on every API call.
user-id HTTP Header — all requests API user ID. Required together with tenant-id on every call.
sourceType: "API" Order.sourceType Marks Midsys as external source of truth. Orders with this value cannot be deleted — lifecycle controlled via orderStatus only.
orderStatus Order field Lifecycle status. 1 = Active, 5 = Cancelled. Controls order activation since deletion is not possible for sourceType "API".
appointmentStatus Appointment field 1 = Active, 5 = Cancelled. To cancel a Patch: set appointmentStatus=5 (do not omit from next PUT).
visitStatus Visit field (Enum) 1=Assigned · 2=Enroute · 3=InProgress · 4=Completed · 5=Cancelled · 6=NotDone · 7=Undone · 8=Unassigned
branch.id Appointment — required Geotab internal Branch ID (Gesellschaft). Fetched once at onboarding via GET /branches. Required on every Appointment and Plan.
orderType.id Order + Appointment — required Geotab internal OrderType ID. Defines duration, cost, icon and scheduling parameters. V1: two types — Midsys-Job and CRM-Adhoc. Constant per tenant.
customerId Appointment — optional External customer number from MEWA (= Customer Number (Kundennummer)). Links Appointment to the Customer record created via POST /customers.
baseline: true PostPlan field Swagger-confirmed flag on POST /plans. Imports the plan as-is from Midsys — no auto-routing or sequence changes. Requires actual start/end times on all visits.
geoCode Address — required GPS coordinates (latitude, longitude). Required field on Appointment address. Source: MEWA CRM. Fallback: GET /common/addresses geocoder (pending confirmation).
arrivalWindowBegin/End Appointment — required Earliest/latest arrival time in "HH:MM" format. Defines the optimization time window. Required on every Appointment.
lockedResource Appointment — optional Reference to a Resource (driver/vehicle) that must be assigned to this visit. Optional — if absent, visitStatus=8 (Unassigned) and Geotab dispatcher assigns later.
planned.sequence Visit.planned field Sequence number for the visit within a resource's route. Used in baseline plans to preserve the Midsys tour sequence.
Rolling 8-day window PUT /orders Sync horizon. MEWA sends all Appointments for the next 8 days daily. Past dates are not updated. Appointments persist even if omitted from a subsequent PUT.
Form Template POST /admin/formTemplates HTML form assigned to one or more OrderTypes and Branches. Driver fills and submits at visit close. Result stored on visit.actuals.formTemplateSubmission. MEWA reads via polling and creates a CRM Service Ticket via webhook. (Phase 9)
CRM Ad-Hoc Job PUT /orders · orderType: CRM_ADHOC Job created in MEWA CRM outside of the Midsys batch. Pushed immediately via PUT /orders with visitStatus=8 (Unassigned). Appointment Engine attempts auto-insert into running day plan. If no slot: stays Unassigned for dispatcher. Status + form submission flow back to CRM. (Phase 10)
Appointment Engine POST /appointmentEngines Geotab component that evaluates whether an Unassigned Appointment can be auto-inserted into a running plan. Checks resource availability, time window and OrderType. If no match: stays visitStatus=8 (Unassigned).
CRM Service Ticket MEWA CRM webhook Ticket created automatically in MEWA CRM when a driver closes a visit with a Form Template submission. Links to Customer Number (Kundennummer) + Appointment ID. Triggered from visit.actuals.formTemplateSubmission during status polling.
